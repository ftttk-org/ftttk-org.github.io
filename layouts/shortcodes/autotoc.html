{{- $page := .Page -}} {{- $content := $page.Content -}} {{- $headings := slice -}} {{- /* 各行を処理して見出しを抽出 */ -}} {{- $lines := split $content "\n" -}} {{- range $lines -}} {{- $line := strings.TrimSpace . -}} {{- if hasPrefix $line "#" -}} {{- $level := 0 -}} {{- $title := "" -}} {{- $anchor := "" -}} {{- /* #の数をカウント */ -}} {{- $chars := split $line "" -}} {{- range $chars -}} {{- if eq . "#" -}} {{- $level = add $level 1 -}} {{- else -}} {{- break -}} {{- end -}} {{- end -}} {{- /* タイトルを抽出 */ -}} {{- $title = strings.TrimSpace (substr $line $level) -}} {{- /* アンカーを生成 */ -}} {{- $anchor = $title | lower | replace " " "-" | replace "　" "-" | replace "(" "" | replace ")" "" | replace "！" "" | replace "？" "" -}} {{- $headings = $headings | append (dict "title" $title "level" $level "anchor" $anchor) -}} {{- end -}} {{- end -}} {{- if $headings -}}

目次

    {{- range $headings -}}
  • {{ .title }}
    • {{- end -}}
{{- else -}}

目次

見出しが見つかりません

{{- end -}}
